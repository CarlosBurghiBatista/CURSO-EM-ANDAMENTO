DESCRIÇÃO DO CÓDIGO BASE DO HTML


1 <-!DOCTYPE html> (Indica que o documento é feito em HTML)

2 <-html lang="pt-br"> (Abertura da tag "html", que engloba todo o documento acrescida do parâmetro "lang=" (Que indica a língua em que foi feito) )

3<-head> (Indica a abertura da área da cabeça do documento, que contém as configurações do site)

4....<-meta charset="UTF-8"> (Configuração dos caracteres, acentuação e etc.. )

5....<-meta name="viewport" content="width=device-width, initial-scale=1.0"> (Escala do documento)

6....<-title>Document /title>( Título do site)

7<-/head> (Fechamento da tag "head")

8<-body>(Tag de abertura do corpo onde ficará o conteúdo do documento)

9<-/body> (Fechamento do corpo do documento)

10<-/html>(Fechamento da tag html que encapsula a cabeça e o corpo do documento)
